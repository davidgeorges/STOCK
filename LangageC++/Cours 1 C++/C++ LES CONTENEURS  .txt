LES CONTENEURS : ( tableau )


les vector : vector<type> nom; ( push_back, pop_back, front() ,back() ,assign() ) // pop = supprimer

les list :  list<type> nom; ( nom.empty() ) // list chainées , chaque element a une valeur et un pointeurs sur le prochain element

les set : set<string> a; // ensemble de chaines de caracteres  (a.clear() pour viter le "conteneurs")

les deque : deque<type> nom; // tableau ou on peut ajouter des éléments aux deux extrémités ( push_back, pop_back, pop_front, push_front)

les stack: stack<type> nom; // une pile , Last in First out , ( push(), top(), pop() )

les queue : queue<type> nom; // file , First in First out , ( push(), front(), pop()
